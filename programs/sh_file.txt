#!/bin/bash
#SBATCH --job-name=nca_main
#SBATCH --partition=work1
#SBATCH --ntasks=1
#SBATCH --cpus-per-task=4
#SBATCH --mem=64gb
#SBATCH --time=24:00:00
#SBATCH --output=/home/%u/nca_job_postings/logs/nca_main_%j.out
#SBATCH --error=/home/%u/nca_job_postings/logs/nca_main_%j.err
#SBATCH --mail-type=END,FAIL

set -euo pipefail

cd ~/nca_job_postings

module purge
module load r/4.5.0

echo "=================================================="
echo "Job started at: $(date)"
echo "Running on node(s): ${SLURM_JOB_NODELIST}"
echo "Job ID: ${SLURM_JOB_ID}"
echo "Submit dir: ${SLURM_SUBMIT_DIR}"
echo "=================================================="

jobperf -record -w -rate 30s &
JOBPERF_PID=$!

srun Rscript ~/nca_job_postings/programs/0d_main_cluster.R

kill ${JOBPERF_PID} 2>/dev/null || true

echo "=================================================="
echo "Job ended at: $(date)"
echo "=================================================="
